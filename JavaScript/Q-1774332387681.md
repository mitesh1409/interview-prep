# Coding Question

Get count for the each character in the given string, for example 'Times of India'.

```javascript
const toi = 'Times of India';

function getCharacterCounts(str, caseSensitive = true) {
    const characterCounts = new Map();

    if (!caseSensitive) {
        str = str.toLowerCase();
    }

    for (char of str) {
        let count = characterCounts.get(char);
        count = !count ? 1 : (count + 1);
        characterCounts.set(char, count);
    }

    return characterCounts;
}

console.log(getCharacterCounts(toi));
```
